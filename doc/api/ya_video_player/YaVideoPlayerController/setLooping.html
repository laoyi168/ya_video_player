Sets whether or not the video should loop after playing once. See also VideoPlayerValue.isLooping.

Implementation

@override
Future<void> setLooping(bool looping) async {
  if (_isFlv) {
    return await YaVideoPlayer._channel
        .invokeMapMethod('setLooping', [_textureId, looping]);
  } else {
    return await super.setLooping(looping);
  }
}
